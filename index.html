              />

              <button onClick={addPack} className="bg-emerald-600 text-white p-2 rounded-lg hover:bg-emerald-700 transition-colors">

                <Plus size={20} />

              </button>

            </div>

          </div>

        </section>

      </div>

    );

  };

  // 3. 旅程總表

  const SummaryTab = () => {

    const [editingDay, setEditingDay] = useState(null);

    const [editForm, setEditForm] = useState({ date: '', title: '' });

    const [confirmDelete, setConfirmDelete] = useState(null);

    const startEdit = (day) => {

      setEditingDay(day.day);

      setEditForm({ date: day.date, title: day.title });

      setConfirmDelete(null);

    };

    const saveEdit = () => {

      setSummary(summary.map(s => s.day === editingDay ? { ...s, date: editForm.date, title: editForm.title } : s));

      setEditingDay(null);

    };

    const requestDelete = (dayNum) => {

      setConfirmDelete(dayNum);

    };

    const executeDelete = (dayNum) => {

      handleDeleteDay(dayNum);

      setConfirmDelete(null);

    };

    return (

      <div className="h-full overflow-y-auto p-6 pb-28 bg-gray-50 animate-fade-in">

        <h2 className="text-2xl font-bold text-gray-800 mb-6 flex items-center gap-2">

          <CalendarDays className="text-indigo-600" />

          旅程總表

        </h2>

       

        {summary.length === 0 ? (

          <div className="text-center text-gray-400 py-10">目前尚無安排天數</div>

        ) : (

          <div className="relative border-l-2 border-indigo-200 ml-4 space-y-6">

            {summary.map((day) => (

              <div key={day.day} className="relative pl-6">

                <div className="absolute w-4 h-4 bg-indigo-500 rounded-full -left-[9px] top-4 ring-4 ring-indigo-100"></div>

               

                {editingDay === day.day ? (

                  // 編輯模式

                  <div className="bg-white p-4 rounded-xl shadow-md border-2 border-indigo-300 animate-fade-in">

                    <div className="mb-3">

                      <label className="text-xs font-bold text-gray-500 mb-1 block">日期 (例如: 11/20 (一))</label>

                      <input

                        type="text"

                        value={editForm.date}

                        onChange={e => setEditForm({...editForm, date: e.target.value})}

                        className="w-full px-3 py-2 border border-gray-200 rounded-lg text-sm focus:outline-none focus:ring-2 focus:ring-indigo-500"

                      />

                    </div>

                    <div className="mb-4">

                      <label className="text-xs font-bold text-gray-500 mb-1 block">行程總結標題</label>

                      <input

                        type="text"

                        value={editForm.title}

                        onChange={e => setEditForm({...editForm, title: e.target.value})}

                        className="w-full px-3 py-2 border border-gray-200 rounded-lg text-sm focus:outline-none focus:ring-2 focus:ring-indigo-500"

                      />

                    </div>

                    <div className="flex gap-2">

                      <button onClick={saveEdit} className="flex-1 bg-indigo-600 text-white py-2 rounded-lg text-sm font-bold">儲存</button>

                      <button onClick={() => setEditingDay(null)} className="flex-1 bg-gray-100 text-gray-600 py-2 rounded-lg text-sm font-bold">取消</button>

                    </div>

                  </div>

                ) : (

                  // 一般顯示模式

                  <div className="bg-white p-4 rounded-xl shadow-sm border border-gray-100 group transition-all hover:border-indigo-200">

                    <div className="flex justify-between items-start mb-1">

                      <div className="text-xs font-bold text-indigo-500 tracking-wider">DAY {day.day} • {day.date}</div>

                      <div className="flex gap-1">

                        <button onClick={() => startEdit(day)} className="text-gray-300 hover:text-indigo-500 p-1 transition-colors">

                          <Pencil size={16} />

                        </button>

                        {confirmDelete === day.day ? (

                          <button onClick={() => executeDelete(day.day)} className="bg-red-500 text-white text-[10px] px-2 py-0.5 rounded font-bold animate-pulse">

                            確認刪除?

                          </button>

                        ) : (

                          <button onClick={() => requestDelete(day.day)} className="text-gray-300 hover:text-red-500 p-1 transition-colors">

                            <Trash2 size={16} />

                          </button>

                        )}

                      </div>

                    </div>

                    <h3 className="text-gray-800 font-medium">{day.title}</h3>

                  </div>

                )}

              </div>

            ))}

          </div>

        )}

        <button

          onClick={handleAddDay}

          className="mt-8 w-full py-3 border-2 border-dashed border-indigo-300 text-indigo-600 rounded-2xl font-bold flex items-center justify-center gap-2 hover:bg-indigo-50 transition-colors"

        >

          <Plus size={20} /> 新增一天行程

        </button>

      </div>

    );

  };

  // 4. 每日細流

  const ItineraryTab = () => {

    const [selectedDay, setSelectedDay] = useState(1);

    const [isAdding, setIsAdding] = useState(false);

    const [newAct, setNewAct] = useState({ time: '', title: '', desc: '', transportToNext: '', transportIcon: 'walk' });

    useEffect(() => {

      if (summary.length > 0 && !summary.find(s => s.day === selectedDay)) {

        setSelectedDay(summary[0].day);

      }

    }, [summary, selectedDay]);

    if (summary.length === 0) {

      return (

        <div className="h-full flex flex-col items-center justify-center bg-gray-50 p-6 text-center">

          <CalendarDays size={48} className="text-gray-300 mb-4" />

          <h2 className="text-xl font-bold text-gray-700 mb-2">尚未建立任何天數</h2>

          <p className="text-gray-500 text-sm mb-6">請先至「總表」頁面新增行程天數，再來規劃每日細流。</p>

          <button onClick={() => setActiveTab('summary')} className="bg-indigo-600 text-white px-6 py-2 rounded-full font-bold">前往總表</button>

        </div>

      );

    }

    const currentSummary = summary.find(s => s.day === selectedDay) || summary[0];

    const currentDayData = itinerary.find(d => d.day === currentSummary.day) || { day: currentSummary.day, activities: [] };

    const handleAddActivity = () => {

      if (!newAct.title || !newAct.time) return;

     

      const updatedItinerary = [...itinerary];

      const dayIndex = updatedItinerary.findIndex(d => d.day === currentSummary.day);

      const newActivityItem = { id: Date.now(), ...newAct };

      if (dayIndex !== -1) {

        updatedItinerary[dayIndex].activities.push(newActivityItem);

        updatedItinerary[dayIndex].activities.sort((a, b) => a.time.localeCompare(b.time));

      } else {

        updatedItinerary.push({ day: currentSummary.day, activities: [newActivityItem] });

      }

      setItinerary(updatedItinerary);

      setNewAct({ time: '', title: '', desc: '', transportToNext: '', transportIcon: 'walk' });

      setIsAdding(false);

    };

    const removeActivity = (dayId, actId) => {

      setItinerary(itinerary.map(d => d.day === dayId ? { ...d, activities: d.activities.filter(a => a.id !== actId) } : d));

    };

    return (

      <div className="flex flex-col h-full bg-gray-50 animate-fade-in">

        <div className="bg-white px-4 py-3 shadow-sm z-10 sticky top-0 flex gap-2 overflow-x-auto no-scrollbar">

          {summary.map(day => (

            <button

              key={day.day}

              onClick={() => setSelectedDay(day.day)}

              className={`flex-shrink-0 px-4 py-2 rounded-full text-sm font-bold transition-all ${

                currentSummary.day === day.day

                  ? 'bg-indigo-600 text-white shadow-md'

                  : 'bg-gray-100 text-gray-500 hover:bg-gray-200'

              }`}

            >

              Day {day.day}

            </button>

          ))}

        </div>

        <div className="flex-1 overflow-y-auto p-4 pb-28">

          <div className="mb-6">

            <h2 className="text-xl font-bold text-gray-800">Day {currentSummary.day} 行程細流</h2>

            <p className="text-gray-500 text-sm mt-1">{currentSummary.date} • {currentSummary.title}</p>

          </div>

          <div className="space-y-0">

            {currentDayData.activities.length === 0 ? (

              <div className="text-center text-gray-400 py-10 bg-white rounded-2xl border border-dashed border-gray-300">

                目前尚無安排行程

              </div>

            ) : (

              currentDayData.activities.map((act, index) => (

                <div key={act.id} className="relative">

                  <div className="bg-white rounded-2xl p-4 shadow-sm border border-gray-100 relative z-10 flex gap-4 group">

                    <div className="flex flex-col items-center">

                      <div className="bg-indigo-50 text-indigo-700 font-bold px-2 py-1 rounded-md text-sm whitespace-nowrap">

                        {act.time}

                      </div>

                    </div>

                    <div className="flex-1 pb-1">

                      <h3 className="text-lg font-bold text-gray-800 mb-1 flex justify-between items-start">

                        {act.title}

                        <button onClick={() => removeActivity(currentSummary.day, act.id)} className="text-gray-300 hover:text-red-500 opacity-0 group-hover:opacity-100 transition-opacity">

                          <Trash2 size={16} />

                        </button>

                      </h3>

                      {act.desc && <p className="text-gray-600 text-sm leading-relaxed mb-2">{act.desc}</p>}

                    </div>

                  </div>

                  {(act.transportToNext || index < currentDayData.activities.length - 1) && (

                    <div className="relative pl-8 py-3 my-1">

                      <div className="absolute left-[39px] top-0 bottom-0 w-0.5 bg-gray-200 z-0"></div>

                      {act.transportToNext && (

                        <div className="relative z-10 flex items-center gap-3 ml-8 bg-gray-100/80 w-fit px-3 py-1.5 rounded-full text-xs text-gray-600 border border-gray-200/50">

                          <span className="text-indigo-500">{renderTransportIcon(act.transportIcon)}</span>

                          {act.transportToNext}

                        </div>

                      )}

                    </div>

                  )}

                </div>

              ))

            )}

          </div>

          {isAdding ? (

            <div className="mt-6 bg-indigo-50 p-4 rounded-2xl border border-indigo-100">

              <h4 className="text-sm font-bold text-indigo-800 mb-3">新增行程節點</h4>

              <div className="space-y-3">

                <div className="flex gap-3">

                  <input type="time" value={newAct.time} onChange={e => setNewAct({...newAct, time: e.target.value})} className="px-3 py-2 rounded-lg border border-gray-200 text-sm flex-shrink-0" required />

                  <input type="text" placeholder="地點 / 標題" value={newAct.title} onChange={e => setNewAct({...newAct, title: e.target.value})} className="flex-1 px-3 py-2 rounded-lg border border-gray-200 text-sm" required />

                </div>

                <textarea placeholder="活動簡介或備註..." value={newAct.desc} onChange={e => setNewAct({...newAct, desc: e.target.value})} className="w-full px-3 py-2 rounded-lg border border-gray-200 text-sm resize-none h-20" />

               

                <div className="bg-white p-3 rounded-lg border border-gray-200">

                  <label className="text-xs font-bold text-gray-500 mb-2 block">前往下一個地點的交通方式 (選填)</label>

                  <div className="flex gap-2">

                    <select value={newAct.transportIcon} onChange={e => setNewAct({...newAct, transportIcon: e.target.value})} className="px-2 py-2 rounded-lg border border-gray-200 text-sm bg-gray-50">

                      <option value="walk">步行</option>

                      <option value="train">火車/地鐵</option>

                      <option value="bus">巴士</option>

                      <option value="car">汽車/計程車</option>

                      <option value="plane">飛機</option>

                    </select>

                    <input type="text" placeholder="例如：搭乘地鐵轉乘公車 (約30分)" value={newAct.transportToNext} onChange={e => setNewAct({...newAct, transportToNext: e.target.value})} className="flex-1 px-3 py-2 rounded-lg border border-gray-200 text-sm" />

                  </div>

                </div>

                <div className="flex gap-2 pt-2">

                  <button onClick={handleAddActivity} className="flex-1 bg-indigo-600 text-white py-2 rounded-lg text-sm font-bold hover:bg-indigo-700 transition-colors">儲存行程</button>

                  <button onClick={() => setIsAdding(false)} className="px-4 py-2 bg-gray-200 text-gray-700 rounded-lg text-sm font-bold hover:bg-gray-300 transition-colors">取消</button>

                </div>

              </div>

            </div>

          ) : (

            <button

              onClick={() => setIsAdding(true)}

              className="mt-6 w-full py-3 border-2 border-dashed border-indigo-300 text-indigo-600 rounded-2xl font-bold flex items-center justify-center gap-2 hover:bg-indigo-50 transition-colors"

            >

              <Plus size={20} /> 新增行程節點

            </button>

          )}

        </div>

      </div>

    );

  };

  // --- 底部導覽列 ---

  const BottomNav = () => (

    <div className="absolute bottom-0 left-0 right-0 bg-white border-t border-gray-200 flex justify-around items-center pb-safe z-50 shadow-[0_-4px_20px_rgba(0,0,0,0.05)]">

      {[

        { id: 'home', icon: Home, label: '首頁' },

        { id: 'prep', icon: ListTodo, label: '準備' },

        { id: 'summary', icon: CalendarDays, label: '總表' },

        { id: 'itinerary', icon: Map, label: '細流' },

      ].map((tab) => (

        <button

          key={tab.id}

          onClick={() => setActiveTab(tab.id)}

          className={`flex flex-col items-center justify-center w-full py-3 space-y-1 transition-colors ${

            activeTab === tab.id ? 'text-indigo-600' : 'text-gray-400 hover:text-gray-600'

          }`}

        >

          <tab.icon size={22} className={activeTab === tab.id ? 'stroke-[2.5px]' : ''} />

          <span className="text-[10px] font-medium">{tab.label}</span>

        </button>

      ))}

    </div>

  );

  return (

    <div className="w-full h-screen bg-gray-100 flex items-center justify-center p-0 sm:p-4 font-sans text-gray-900">

      <div className="w-full h-full sm:max-w-[400px] sm:h-[850px] bg-white sm:rounded-[2.5rem] sm:shadow-2xl overflow-hidden relative flex flex-col border-0 sm:border-[8px] sm:border-gray-900">

        <div className="hidden sm:block absolute top-0 inset-x-0 h-6 bg-gray-900 rounded-b-3xl w-40 mx-auto z-50"></div>

       

        <div className="flex-1 overflow-hidden relative bg-gray-50">

          {activeTab === 'home' && <HomeTab />}

          {activeTab === 'prep' && <PrepTab />}

          {activeTab === 'summary' && <SummaryTab />}

          {activeTab === 'itinerary' && <ItineraryTab />}

        </div>

        <BottomNav />

      </div>

    </div>

  );

}
